int Func factorial (int n, str text) {      
    my int result;
    OUT text;
    result = 1;
    my int i;
    Cycle 1...n{
      result=result*i;
    }
    return text;
}


bool Func is_even(int num) {
    my int check;
    check = num & 1;
    return check == 0;
}
